// src/renderer/src/plugins/context/index.tsx const plugin: Plugin<TranscriptStats> = { id: 'context', title: 'Context Window', icon: '🧠', intervalMs: 2000, defaultEnabled: true, poll: (gt) => gt.transcript(), // read live state render: (d) => ( <Card icon="🧠" title="Context Window"> <Big value={`${d.contextPct.toFixed(1)}%`} /> <Gauge pct={d.contextPct} /> </Card> ), } export default plugin
